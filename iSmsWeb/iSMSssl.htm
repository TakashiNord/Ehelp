##################### АВТОРИЗАЦИЯ С ПОМОЩЬЮ КЛИЕНТСКИХ SSL СЕРТИФИКАТОВ #######################
###############################################################################################

# Перед началом необходимо:
# установить программу openssl, содержащуюся в пакете openssl
apt-get update
apt-get install openssl

# включить модуль apache2 mod_ssl
a2enmod ssl

# создать каталог /etc/apache2/ssl и перейти в него
mkdir -p /etc/apache2/ssl && cd /etc/apache2/ssl

###############################################################################################
########### 1. Создание собственного самоподписанного доверенного сертификата. (CA) ###########
###############################################################################################
openssl req -new -newkey rsa:1024 -nodes -keyout ca.key -x509 \
-subj /C=RU/ST=Msk/L=Msk/O=EMA/OU=ORPR/CN=admin/emailAddress=usr@dom.ru \
-out ca.crt

###############################################################################################
########################### 2. Создание клиентских сертификатов ###############################
###############################################################################################
# Создание конфиг. файла ca.config для подписи запроса на сертификат (CSR)
# с помощью доверенного сертификата (CA)
# со следующим содержимым
[ ca ]
default_ca        = CA_CLIENT

[ CA_CLIENT ]
dir                    = ./db
certs                  = $dir/certs
new_certs_dir          = $dir/newcerts
database               = $dir/index.txt
serial                 = $dir/serial
certificate            = ./ca.crt
private_key            = ./ca.key
default_days           = 3660
default_crl_days       = 7
default_md             = md5
policy                 = policy_anything

[ policy_anything ]
countryName            = optional
stateOrProvinceName    = optional
localityName           = optional
organizationName       = optional
organizationalUnitName = optional
commonName             = supplied
emailAddress           = optional

# Создание каталогов для сертификатов и служебных файлов
mkdir -p db/certs && mkdir -p db/newcerts && touch db/index.txt && echo "01" > db/serial

# Создание клиентского закрытого ключа и запроса на сертификат (CSR)
openssl req -new -newkey rsa:1024 -nodes -keyout client01.key \
-subj /C=RU/ST=Msk/L=Msk/O=KOLENERGO/OU=SES/CN=isms/emailAddress=admin@ema.ru \
-out client01.csr

# Подпись запроса на сертификат (CSR) с помощью доверенного сертификата (CA).
openssl ca -config ca.config -in client01.csr -out client01.crt -batch

# Подготовка данных для передачи клиенту.
# параметр pass указывает пароль, который необходимо будет ввести при импорте сертификата в браузере
openssl pkcs12 -export -in client01.crt -inkey client01.key \
-certfile ca.crt -out client01.p12 -passout pass:iSmSPassw0rd

###############################################################################################
#################################### 3. Настройка веб-сервера. ################################
###############################################################################################
# Найдите в конфиге веб-сервера apache2 секцию VirtualHost отвечающую за iSMS
# (обычно это файл /etc/apache2/sites-available/wasutp.conf)
# и сконфигурируйте в соостветствии с конфигом ниже:
<VirtualHost *:80>
        ServerName wasutp.loc
        ServerAlias www.wasutp.loc
        ServerAdmin spendlively@mail.ru
        DocumentRoot /home/www-data/www/htdocs/WASUTP/public/
        ErrorLog ${APACHE_LOG_DIR}/error.log
        CustomLog ${APACHE_LOG_DIR}/access.log combined
        <Directory /home/www-data/www/htdocs/WASUTP/public/>
                AllowOverride All
                Options +FollowSymLinks +Indexes
                Order deny,allow
            Require all granted # для apache2 >= 2.4
#           Allow from all # для apache2 < 2.4
        </directory>
</VirtualHost>

<VirtualHost *:443>
    ServerName wasutp.loc
    ServerAlias www.wasutp.loc
    DocumentRoot /home/www-data/www/htdocs/WASUTP/public/
    <Directory /home/www-data/www/htdocs/WASUTP/public/>
        AllowOverride All
        Options +FollowSymLinks
        Order Deny,Allow
      Require all granted # для apache2 >= 2.4
#     Allow from all # для apache2 < 2.4
        SSLVerifyClient require
    </Directory>
        SSLEngine on
        SSLVerifyClient require
        SSLCertificateKeyFile /etc/apache2/ssl/ca.key
        SSLCertificateFile    /etc/apache2/ssl/ca.crt
        SSLCACertificateFile /etc/apache2/ssl/ca.crt
</VirtualHost>
